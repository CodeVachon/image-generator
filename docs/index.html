Image Generator

Image Generator

Install as an NPM Module

add this to your .npmrc file.

@christophervachon:registry https://npm.christophervachon.com

and Download using NPM Scope @christophervachon

npm install @christophervachon/image-generator

Usage

 
import defaultImage from "@christophervachon/image-generator";

export default ImageNode = () => (<img src={ defaultImage("Test String") } alt="Test String" />);

Options

 
interface ExpectedConfigObject {
    fontSize?: number, // font size in PX: default 40
    font?: string, // font to use: default Arial
    textColor?: string, // text color: default #999
    fillColor?: string | Array<string>, // background color(s): default #ccc
    gradientType?: "radial" | "linear", // if fillColor is an Array, set a gradient type: default radial
    borderColor?: string, // Color of the Border: Default #999
    borderWidth?: number, // Width of the Border in PX: default 4
    width?: number, // Width of the Image Generated in PX: default 400
    height?: number, // Height of the Image Generated in PX: default 400
} // close ExpectedConfigObject
